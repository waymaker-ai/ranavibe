```typescript
import { z } from 'zod';
import yaml from 'js-yaml';

// Security rule severity levels
const SeverityLevel = z.enum(['critical', 'high', 'medium', 'low', 'info']);

// Security levels
const SecurityLevel = z.enum(['minimal', 'standard', 'enterprise', 'paranoid']);

// Individual security rule
const SecurityRuleSchema = z.object({
  id: z.string().min(1),
  severity: SeverityLevel,
  pattern: z.string().min(1),
  detect: z.string().optional(),
  fix: z.string().optional(),
});

// Individual compliance rule
const ComplianceRuleSchema = z.object({
  id: z.string().min(1),
  standard: z.string().min(1),
  pattern: z.string().min(1),
  detect: z.string().optional(),
  fix: z.string().optional(),
});

// Context section
const ContextSchema = z.object({
  description: z.string().min(1),
  architecture: z.string().optional(),
  frameworks: z.array(z.string()).optional(),
  structure: z.record(z.string(), z.string()).optional(),
  conventions: z.array(z.string()).optional(),
});

// Security section
const SecuritySchema = z.object({
  level: SecurityLevel,
  rules: z.array(SecurityRuleSchema).optional(),
}).optional();

// Compliance section
const ComplianceSchema = z.object({
  presets: z.array(z.string()).optional(),
  custom: z.array(ComplianceRuleSchema).optional(),
}).optional();

// Workflow section
const WorkflowSchema = z.object({
  testing: z.string().optional(),
  linting: z.string().optional(),
  building: z.string().optional(),
  deploying: z.string().optional(),
  branching: z.string().optional(),
  'pr-rules': z.array(z.string()).optional(),
}).optional();

// Governance section
const GovernanceSchema = z.object({
  models: z.object({
    preferred: z.array(z.string()).optional(),
    forbidden: z.array(z.string()).optional(),
    reasoning: z.string().optional(),
  }).optional(),
  cost: z.object({
    budget: z.number().positive().optional(),
    alerts: z.array(z.number().min(0).max(100)).optional(),
  }).optional(),
  quality: z.record(z.string(), z.union([z.number(), z.boolean(), z.string()])).optional(),
}).optional();

// Main rules schema
export const RulesSchema = z.object({
  version: z.literal('3.0'),
  name: z.string().min(1),
  context: ContextSchema,
  security: SecuritySchema,
  compliance: ComplianceSchema,
  workflow: WorkflowSchema,
  governance: GovernanceSchema,
});

export type RulesConfig = z.infer<typeof RulesSchema>;
export type SecurityRule = z.infer<typeof SecurityRuleSchema>;
export type ComplianceRule = z.infer<typeof ComplianceRuleSchema>;

/**
 * Parse a YAML string into a validated RulesConfig.
 * Throws if the YAML is invalid or doesn't match the schema.
 */
export function parseRules(yamlString: string): RulesConfig {
  const raw = yaml.load(yamlString);
  return RulesSchema.parse(raw);
}
```

```typescript
  '@supabase/supabase-js': 'supabase',
  'firebase': 'firebase',
  'mongoose': 'mongoose',
  'typeorm': 'typeorm',
};

export function detectFrameworks(repoPath: string): string[] {
  const pkgPath = path.join(repoPath, 'package.json');
  if (!fs.existsSync(pkgPath)) return [];

  try {
    const pkg = JSON.parse(fs.readFileSync(pkgPath, 'utf-8'));
    const allDeps = {
      ...pkg.dependencies,
      ...pkg.devDependencies,
    };

    const detected: string[] = [];
    for (const [dep, framework] of Object.entries(FRAMEWORK_INDICATORS)) {
      if (allDeps[dep]) {
        detected.push(framework);
      }
    }
    return [...new Set(detected)];
  } catch {
    return [];
  }
}
```

```typescript
import fs from 'fs';
import path from 'path';

const DIRECTORY_DESCRIPTIONS: Record<string, string> = {
  'src/app': 'Next.js app router pages',
  'src/pages': 'Next.js pages router',
  'app': 'Next.js app router pages',
  'pages': 'Pages directory',
  'src/components': 'React components',
  'src/lib': 'Shared utilities and helpers',
  'src/utils': 'Utility functions',
  'src/hooks': 'React hooks',
  'src/services': 'Service layer',
  'src/api': 'API routes or handlers',
  'src/types': 'TypeScript type definitions',
  'src/styles': 'Stylesheets',
  'src/config': 'Configuration files',
  'src/middleware': 'Middleware functions',
  'src/models': 'Data models',
  'src/store': 'State management',
  'lib': 'Shared library code',
  'components': 'UI components',
  'utils': 'Utility functions',
  'hooks': 'React hooks',
  'tests': 'Test files',
  '__tests__': 'Test files',
  'prisma': 'Prisma database schema and migrations',
  'public': 'Static assets',
  'scripts': 'Build and utility scripts',
  'docs': 'Documentation',
  'packages': 'Monorepo packages',
};

export function detectStructure(repoPath: string): Record<string, string> {
  const structure: Record<string, string> = {};

  function scan(dir: string, relativePath: string = ''): void {
    try {
      const entries = fs.readdirSync(dir, { withFileTypes: true });
      for (const entry of entries) {
        if (!entry.isDirectory()) continue;
        if (entry.name.startsWith('.') || entry.name === 'node_modules' || entry.name === 'dist') continue;

        const rel = relativePath ? `${relativePath}/${entry.name}` : entry.name;

        // Check if this matches a known pattern
        if (DIRECTORY_DESCRIPTIONS[rel]) {
          structure[rel] = DIRECTORY_DESCRIPTIONS[rel];
        }

        // Check for monorepo package directories
        if (rel.startsWith('packages/') && rel.split('/').length === 2) {
          const pkgJson = path.join(dir, entry.name, 'package.json');
          if (fs.existsSync(pkgJson)) {
            try {
              const pkg = JSON.parse(fs.readFileSync(pkgJson, 'utf-8'));
              structure[rel] = pkg.description || `Package: ${entry.name}`;
            } catch {
              structure[rel] = `Package: ${entry.name}`;
            }
          }
        }

        // Recurse into src/ and common directories (max 3 levels)
        if (rel.split('/').length < 3) {
          scan(path.join(dir, entry.name), rel);
        }
      }
    } catch {
      // Permission denied or other FS error
    }
  }

  scan(repoPath);
  return structure;
}
```

```typescript
import fs from 'fs';
import path from 'path';

export interface DependencyInfo {
  packageManager: 'npm' | 'yarn' | 'pnpm' | 'bun' | 'unknown';
  key: string[];
  devTools: string[];
}

const KEY_DEPS = [
  'prisma', 'drizzle-orm', 'mongoose', 'typeorm', 'sequelize',
  'zod', 'yup', 'joi',
  'axios', 'ky', 'got',
  'redis', 'ioredis',
  '@supabase/supabase-js', 'firebase', '@aws-sdk/client-s3',
  'stripe', 'twilio',
  'passport', 'next-auth', '@clerk/nextjs', '@auth/core',
  'socket.io', 'ws',
  'sharp', 'jimp',
  'bull', 'bullmq',
  'winston', 'pino',
];

export function detectDependencies(repoPath: string): DependencyInfo {
  const result: DependencyInfo = {
    packageManager: 'unknown',
    key: [],
    devTools: [],
  };

  // Detect package manager from lockfile
  if (fs.existsSync(path.join(repoPath, 'pnpm-lock.yaml'))) {
    result.packageManager = 'pnpm';
  } else if (fs.existsSync(path.join(repoPath, 'bun.lockb')) || fs.existsSync(path.join(repoPath, 'bun.lock'))) {
    result.packageManager = 'bun';
  } else if (fs.existsSync(path.join(repoPath, 'yarn.lock'))) {
    result.packageManager = 'yarn';
  } else if (fs.existsSync(path.join(repoPath, 'package-lock.json'))) {
    result.packageManager = 'npm';
  }

  // Parse package.json for dependencies
  const pkgPath = path.join(repoPath, 'package.json');
  if (!fs.existsSync(pkgPath)) return result;

  try {
    const pkg = JSON.parse(fs.readFileSync(pkgPath, 'utf-8'));
    const deps = Object.keys(pkg.dependencies || {});
    const devDeps = Object.keys(pkg.devDependencies || {});

    result.key = deps.filter((d) => KEY_DEPS.includes(d));
    result.devTools = devDeps.filter((d) =>
      ['vitest', 'jest', 'playwright', 'cypress', 'eslint', 'prettier', 'typescript', 'tsup', 'esbuild', 'vite', 'webpack'].includes(d)
    );
  } catch {
    // Invalid package.json
  }

  return result;
}
```

```typescript
import fs from 'fs';
import path from 'path';

export function detectConventions(repoPath: string): string[] {
  const conventions: string[] = [];

  // Check for TypeScript
  if (fs.existsSync(path.join(repoPath, 'tsconfig.json'))) {
```

```typescript
import type { RulesConfig } from '../rules/schema';

export type CompilerTarget = 'claude-md' | 'cursorrules' | 'copilot' | 'continue' | 'agents-md' | 'gemini-md';

export abstract class BaseCompiler {
  abstract readonly target: CompilerTarget;
  abstract compile(rules: RulesConfig): string;

  protected renderContext(rules: RulesConfig): string {
    const lines: string[] = [];
    const ctx = rules.context;

    lines.push(`Project: ${rules.name}`);
    lines.push(`Description: ${ctx.description}`);

    if (ctx.architecture) lines.push(`Architecture: ${ctx.architecture}`);
    if (ctx.frameworks?.length) lines.push(`Frameworks: ${ctx.frameworks.join(', ')}`);

    if (ctx.structure && Object.keys(ctx.structure).length > 0) {
      lines.push('', 'Project Structure:');
      for (const [dir, desc] of Object.entries(ctx.structure)) {
        lines.push(`  ${dir} - ${desc}`);
      }
    }

    if (ctx.conventions?.length) {
      lines.push('', 'Conventions:');
      for (const conv of ctx.conventions) {
        lines.push(`- ${conv}`);
      }
    }

    return lines.join('\n');
  }

  protected renderSecurity(rules: RulesConfig): string {
    if (!rules.security) return '';

    const lines: string[] = [];
    lines.push(`Security Level: ${rules.security.level}`);

    if (rules.security.rules?.length) {
      lines.push('', 'Security Rules:');
      for (const rule of rules.security.rules) {
        lines.push(`- [${rule.severity.toUpperCase()}] ${rule.pattern}`);
        if (rule.fix) lines.push(`  Fix: ${rule.fix}`);
      }
    }

    return lines.join('\n');
  }

  protected renderCompliance(rules: RulesConfig): string {
    if (!rules.compliance) return '';

    const lines: string[] = [];

    if (rules.compliance.presets?.length) {
      lines.push(`Compliance Presets: ${rules.compliance.presets.join(', ').toUpperCase()}`);
    }

    if (rules.compliance.custom?.length) {
      lines.push('', 'Compliance Rules:');
      for (const rule of rules.compliance.custom) {
        lines.push(`- [${rule.standard}] ${rule.pattern}`);
      }
    }

    return lines.join('\n');
  }

  protected renderWorkflow(rules: RulesConfig): string {
    if (!rules.workflow) return '';

    const lines: string[] = [];
    const wf = rules.workflow;

    if (wf.testing) lines.push(`Testing: ${wf.testing}`);
    if (wf.linting) lines.push(`Linting: ${wf.linting}`);
    if (wf.building) lines.push(`Building: ${wf.building}`);
    if (wf.deploying) lines.push(`Deploying: ${wf.deploying}`);
    if (wf.branching) lines.push(`Branching: ${wf.branching}`);

    if (wf['pr-rules']?.length) {
      lines.push('', 'PR Rules:');
      for (const rule of wf['pr-rules']) {
        lines.push(`- ${rule}`);
      }
    }

    return lines.join('\n');
  }

  protected renderGovernance(rules: RulesConfig): string {
    if (!rules.governance) return '';

    const lines: string[] = [];
    const gov = rules.governance;

    if (gov.models?.preferred?.length) {
      lines.push(`Preferred Models: ${gov.models.preferred.join(', ')}`);
    }
    if (gov.models?.forbidden?.length) {
      lines.push(`Forbidden Models: ${gov.models.forbidden.join(', ')}`);
    }
    if (gov.models?.reasoning) {
      lines.push(`Model Guidance: ${gov.models.reasoning}`);
    }

    return lines.join('\n');
  }
}
```

```typescript
import type { RulesConfig } from '../rules/schema';
import type { CompilerTarget } from './base';
import { ClaudeMDCompiler } from './claude-md';
import { CursorRulesCompiler } from './cursorrules';
import { CopilotCompiler } from './copilot';
import { ContinueCompiler } from './continue-rules';
import { AgentsMDCompiler } from './agents-md';
import { GeminiMDCompiler } from './gemini-md';

const compilers = {
  'claude-md': new ClaudeMDCompiler(),
  'cursorrules': new CursorRulesCompiler(),
  'copilot': new CopilotCompiler(),
  'continue': new ContinueCompiler(),
  'agents-md': new AgentsMDCompiler(),
  'gemini-md': new GeminiMDCompiler(),
} as const;

const ALL_TARGETS: CompilerTarget[] = Object.keys(compilers) as CompilerTarget[];

/**
 * Compile rules to a specific target format.
 */
export function compileTarget(rules: RulesConfig, target: CompilerTarget): string {
  const compiler = compilers[target];
  if (!compiler) {
    throw new Error(`Unknown compiler target: ${target}. Available: ${ALL_TARGETS.join(', ')}`);
  }
  return compiler.compile(rules);
}

/**
 * Compile rules to all (or selected) targets.
 * Returns a map of target -> compiled output.
 */
export function compileRules(
  rules: RulesConfig,
  targets?: CompilerTarget[],
): Record<string, string> {
  const selected = targets || ALL_TARGETS;
  const results: Record<string, string> = {};
  for (const target of selected) {
    results[target] = compileTarget(rules, target);
  }
  return results;
}

export type { CompilerTarget } from './base';
export { BaseCompiler } from './base';
```

```typescript
import type { RulesConfig, SecurityRule, ComplianceRule } from './schema';
import { getPreset } from './presets/index';

/**
 * Merge preset rules into a base RulesConfig.
 * Custom rules take precedence over preset rules (matched by id).
 */
export function mergeWithPresets(base: RulesConfig): RulesConfig {
  const presetNames = base.compliance?.presets || [];
  if (presetNames.length === 0) return base;

  // Collect all preset rules
  const presetSecurityRules: SecurityRule[] = [];
  const presetComplianceRules: ComplianceRule[] = [];

  for (const name of presetNames) {
    const preset = getPreset(name);
    presetSecurityRules.push(...preset.security.rules);
    presetComplianceRules.push(...preset.compliance.rules);
  }

  // Get existing custom rules
  const customSecurityRules = base.security?.rules || [];
  const customComplianceRules = base.compliance?.custom || [];

  // Custom rules override preset rules with same id
  const customSecurityIds = new Set(customSecurityRules.map((r) => r.id));
  const customComplianceIds = new Set(customComplianceRules.map((r) => r.id));

  const mergedSecurityRules = [
    ...customSecurityRules,
    ...presetSecurityRules.filter((r) => !customSecurityIds.has(r.id)),
  ];

  const mergedComplianceRules = [
    ...customComplianceRules,
    ...presetComplianceRules.filter((r) => !customComplianceIds.has(r.id)),
  ];

  return {
    ...base,
    security: {
      level: base.security?.level || 'standard',
      rules: mergedSecurityRules.length > 0 ? mergedSecurityRules : undefined,
    },
    compliance: {
      ...base.compliance,
      custom: mergedComplianceRules.length > 0 ? mergedComplianceRules : undefined,
    },
  };
}
```

```typescript
import yaml from 'js-yaml';
import { scanRepository } from './scanner/index';
import { RulesSchema, type RulesConfig } from './rules/schema';
import { mergeWithPresets } from './rules/merger';
import { compileRules, type CompilerTarget } from './compiler/index';

export interface GenerateOptions {
  name: string;
  description: string;
  presets?: string[];
  targets?: CompilerTarget[];
  securityLevel?: 'minimal' | 'standard' | 'enterprise' | 'paranoid';
}

export interface GenerateResult {
  rules: RulesConfig;
  rulesYaml: string;
  compiled: Record<string, string>;
}

/**
 * Scan a repository and generate governance rules + compiled tool configs.
 */
export function generateRules(repoPath: string, options: GenerateOptions): GenerateResult {
  // Step 1: Scan the repository
  const scan = scanRepository(repoPath);

  // Step 2: Build base rules from scan
  const baseRules: RulesConfig = {
    version: '3.0',
    name: options.name,
    context: {
      description: options.description,
      architecture: scan.architecture,
      frameworks: scan.frameworks.length > 0 ? scan.frameworks : undefined,
      structure: Object.keys(scan.structure).length > 0 ? scan.structure : undefined,
      conventions: scan.conventions.length > 0 ? scan.conventions : undefined,
    },
    security: {
      level: options.securityLevel || 'standard',
    },
    compliance: options.presets?.length
      ? { presets: options.presets }
      : undefined,
    workflow: {
      testing: scan.dependencies.devTools.includes('vitest')
        ? `${scan.dependencies.packageManager} test`
        : scan.dependencies.devTools.includes('jest')
          ? `${scan.dependencies.packageManager} test`
          : undefined,
      linting: scan.dependencies.devTools.includes('eslint')
        ? `${scan.dependencies.packageManager} lint`
        : undefined,
      building: `${scan.dependencies.packageManager} build`,
    },
  };

  // Validate
  const validated = RulesSchema.parse(baseRules);

  // Step 3: Merge with presets
  const merged = mergeWithPresets(validated);

  // Step 4: Compile to targets
  const compiled = compileRules(merged, options.targets);

  // Step 5: Serialize to YAML
  const rulesYaml = yaml.dump(merged, { lineWidth: 120, quotingType: '"' });

  return {
    rules: merged,
    rulesYaml: `# RANA Rules Configuration\n# Generated by RANA v3.0\n# Docs: https://rana.cx/docs/rules\n\nversion: "3.0"\n${rulesYaml.replace(/^version:.*\n/, '')}`,
    compiled,
  };
}
```

```typescript
        console.log(chalk.red('  [FAIL] .rana/rules.yml not found — run `rana init`'));
        issues++;
      }

      // Check for generated configs
      const configs: Record<string, string> = {
        'CLAUDE.md': 'Claude Code',
        '.cursorrules': 'Cursor',
        '.github/copilot-instructions.md': 'GitHub Copilot',
        'AGENTS.md': 'Universal agents',
      };

      for (const [file, tool] of Object.entries(configs)) {
        if (fs.existsSync(path.join(cwd, file))) {
          console.log(chalk.green(`  [PASS] ${file} exists (${tool})`));
        } else {
          console.log(chalk.yellow(`  [WARN] ${file} missing (${tool}) — run \`rana sync\``));
        }
      }

      // Check for package.json
      if (fs.existsSync(path.join(cwd, 'package.json'))) {
        console.log(chalk.green('  [PASS] package.json found'));
      } else {
        console.log(chalk.yellow('  [WARN] No package.json — limited scanning capabilities'));
      }

      // Check .gitignore for .rana/
      const gitignorePath = path.join(cwd, '.gitignore');
      if (fs.existsSync(gitignorePath)) {
        const gitignore = fs.readFileSync(gitignorePath, 'utf-8');
        if (!gitignore.includes('.rana/')) {
          console.log(chalk.yellow('  [WARN] .rana/ not in .gitignore — consider adding it'));
        }
      }

      console.log('');
      if (issues === 0) {
        console.log(chalk.bold.green('  Everything looks good!\n'));
      } else {
        console.log(chalk.bold.yellow(`  ${issues} issue(s) found. See above.\n`));
      }
    });
}
```
